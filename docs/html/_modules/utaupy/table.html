[ドキュメント]def load(path):
    """テーブルを読み取ってインスタンス生成"""
    # ファイル読み取り
    with open(path) as f:
        l = [v.strip().split() for v in f.readlines()]
    # 辞書にする
    d = {}
    for v in l:
        d[v[0]] = v[1:]
    return d
